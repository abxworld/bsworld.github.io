import java.util.*;


public class Solution {
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     * 计算两个数之和
     * @param s string字符串 表示第一个整数
     * @param t string字符串 表示第二个整数
     * @return string字符串
     */
    public String solve (String s, String t) {
        char[] sa = s.toCharArray();
        char[] ta = t.toCharArray();
        StringBuilder res = new StringBuilder();
        int maxLenght = sa.length >= ta.length ? sa.length : ta.length;
        int sl = sa.length -1;
        int tl = ta.length -1;
        int beyond = 0;
        while(maxLenght >0){
            int sr = 0;
            int tr = 0;
            if(sl >= 0){
                sr = Integer.parseInt(sa[sl] + "");
            }
            if(tl >=0){
                tr = Integer.parseInt(ta[tl] + "");
            }
            int curR = (tr + sr  + beyond) % 10;
            beyond = (tr + sr  + beyond) / 10;
            res.append(curR);
            sl--;
            tl--;
            maxLenght--;
        }
        if(beyond > 0){
            res.append(beyond);
        }
        return res.reverse().toString();
    }
}
