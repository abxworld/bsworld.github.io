package hello;

/**
 * program: limiter 
 * author: bsworld.xie 
 * create: 2021-05-14 11:34 
 * description: 
 */
public class MulipyThreadPrint {
	/** * 共享数据类。 * 多个线程访问的数据区，注意数据的可见性 */
    public static class ShareData {
      	/** * 下一个要打印数字的线程Id * 保证可见性 */
        private volatile int printThreadId;
			/** * 下一个需要打印的数字 * 保证可见性 */
        private volatile int nextToPrint;
				/** * 最大打印值 */
        private int maxPrint;

			/** * 线程数 */
        private int maxThreadNum;

        public ShareData(int printThreadId, int maxPrint, int nextToPrint, int maxThreadNum) {
            this.printThreadId = printThreadId;
            this.maxPrint = maxPrint;
            this.nextToPrint = nextToPrint;
            this.maxThreadNum = maxThreadNum;
        }
    }
	/** * 打印线程类。 * 每个线程有一个id，当shareDate中的printThreadId等于id时，代表到了自己打印的时机 * 通过判断是否属于自己打印的时机，有两种选择： * 1. 打印，将打印的数字+1，更新下一个要打印数字的编号，唤醒其他所有的线程 * 2. 等待自己被唤醒 */

    public static class PrintRunnable implements Runnable {
        private int id;

        private ShareData shareData;

        public PrintRunnable(int id, ShareData shareData) {
            this.id = id;
            this.shareData = shareData;
        }

        @Override
        public void run() {
            while (shareData.nextToPrint <= shareData.maxPrint) {
                synchronized (shareData) {
                    if (id == shareData.printThreadId) {
                        System.out.println("ID为:" + shareData.printThreadId + "值为" + shareData.nextToPrint);
                      //将打印的数字加1
                        shareData.nextToPrint = shareData.nextToPrint + 1;
                       //更新下一个要打印的线程编号
                        shareData.printThreadId = shareData.printThreadId + 1 <= shareData.maxThreadNum ? shareData.printThreadId + 1 : 1;
                        //这里是唤醒所有的线程，唤起一个notify()会出现死锁
                        shareData.notifyAll();
                    } else {
                        try {                      
                            shareData.wait();
                        } catch (InterruptedException e) {
                            e.printStackTrace();
                        }
                    }
                }
            }
        }
    }

		/** * 主类。 * 负责创建共享数据和线程，并启动多个线程 */
    public static void main(String[] args) {
      //共享数据，同时也是锁住的对象
        ShareData shareData = new ShareData(1, 1000, 1, 3);
       //创建多个线程，分别开始打印任务
        for (int i = 1; i <= 3; i++) {
            new Thread(new PrintRunnable(i, shareData)).start();
        }
    }
}
